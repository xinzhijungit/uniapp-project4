    height INT COMMENT '身高（cm）',
    occupation VARCHAR(50) COMMENT '职业',
    workplace VARCHAR(100) COMMENT '服务处所',
    relation_to_head VARCHAR(20) NOT NULL COMMENT '与户主关系（户主/配偶/子/女/孙子等）',
    add_reason VARCHAR(50) NOT NULL COMMENT '新增原因（出生申报/迁入/立户/夫妻投靠等）',
    add_date DATE NOT NULL COMMENT '登记日期（该人加入该户籍的日期）',
    previous_address VARCHAR(100) COMMENT '何时由何地迁来本市/县（原住址）',
    move_in_address VARCHAR(100) COMMENT '何时由何地迁来本址（具体到门牌）',
    page_number INT COMMENT '户口本所在页码',
    registrar VARCHAR(50) COMMENT '承办人签章（登记该人的民警）',
    -- 户籍信息字段（允许重复，与人员记录存储在一起）
    household_id VARCHAR(20) NOT NULL COMMENT '户籍编号（如 H326，允许重复）',
    household_number VARCHAR(20) NOT NULL COMMENT '户号（户口本上的户号）',
    address VARCHAR(100) NOT NULL COMMENT '户籍地址（标准门牌）',
    household_type VARCHAR(20) DEFAULT '家庭户口' COMMENT '户别（家庭户口/集体户口）',
    created_date DATE NOT NULL COMMENT '建户时间（该户籍首次建立日期）',
    last_issued_date DATE COMMENT '户口本最近签发日期',
    issuing_authority VARCHAR(100) COMMENT '签发机关（派出所）',
    issuing_officer VARCHAR(50) COMMENT '承办民警（签章姓名）',
    official_seal VARCHAR(50) COMMENT '户口专用章名称（如“西安市公安局高新分局细柳派出所户口专用章”）',
    head_id_card VARCHAR(18) NOT NULL COMMENT '户主身份证号'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='人员与户籍合并表';


19、案底库（公安违法/刑事犯罪案底库表（包含在逃人员））
CREATE TABLE criminal_record_db (
    record_id INT PRIMARY KEY AUTO_INCREMENT COMMENT '记录唯一编号（主键）',
    id_card VARCHAR(18) NOT NULL COMMENT '身份证号（18位）',
    name VARCHAR(50) NOT NULL COMMENT '姓名',
    gender CHAR(1) CHECK (gender IN ('M', 'F')) COMMENT '性别：M男 / F女',
    birth_date DATE COMMENT '出生日期',
    household_addr VARCHAR(200) COMMENT '户籍地址',
    case_number VARCHAR(50) COMMENT '案件编号',
    crime_type VARCHAR(100) COMMENT '罪名/违法行为',
    crime_detail TEXT COMMENT '犯罪事实简述',
    arrest_date DATE COMMENT '逮捕/拘留日期',
    sentence_date DATE COMMENT '判决日期（在逃人员为空）',
    sentence_months INT COMMENT '刑期（月），在逃人员为NULL',
    prison_status VARCHAR(20) CHECK (prison_status IN ('已服刑', '在逃', '通缉')) COMMENT '服刑状态：已服刑/在逃/通缉',
    escape_notice_no VARCHAR(50) COMMENT '在逃编号/通缉令号（无则为空）',
    created_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '入库时间'
) COMMENT='公安违法/刑事犯罪案底库表（包含在逃人员）';
